215 // If set to vertical mode, the increment is 32, so it skips

225 {
226 // LAUGHTALE_ENGINR_LOG_INFO("ppu write:" << datatoHexString(addr, 4) << ", " << datatoHexString(data, 2));
227
228 addr = addr & 0x0007;

230 switch (addr)
231 {
232 case 0x0000: // Control
233 control.reg = data;
234 tram_addr.nametable_x = control.nametable_x;
235 tram_addr.nametable_y = control.nametable_y;
236 break;
237 case 0x0001: // Mask
238 mask.reg = data;
239 break;
240 case 0x0002: // Status
241 break;
242 case 0x0003: // OAM Address
243 break;
244 case 0x0004: // OAM Data
245 break;
246 case 0x0005: // Scroll
247 if (address_latch == 0)
248 {
249 // First write to scroll register contains X offset in pixel space
250 // which we split into coarse and fine x values
251 fine_x = data & 0x07;
252 tram_addr.coarse_x = data >> 3;
253 address_latch = 1;
254 }
255 else
256 {
257 // First write to scroll register contains Y offset in pixel space
258 // which we split into coarse and fine Y values
259 tram_addr.fine_y = data & 0x07;
260 tram_addr.coarse_y = data >> 3;
261 address_latch = 0;
262 }
263 break;
264 case 0x0006: // PPU Address
265 if (address_latch == 0)
266 {
267 // PPU address bus can be accessed by CPU via the ADDR and DATA
268 // registers. The fisrt write to this register latches the high byte
269 // of the address, the second is the low byte. Note the writes
270 // are stored in the tram register...
271 tram_addr.reg = (uint16_t)((data & 0x3F) << 8) | (tram_addr.reg & 0x00FF);
272 address_latch = 1;
273 }
274 else
275 {
276 // ...when a whole address has been written, the internal vram address
277 // buffer is updated. Writing to the PPU is unwise during rendering
278 // as the PPU will maintam the vram address automatically whilst
279 // rendering the scanline position.
280 tram_addr.reg = (tram_addr.reg & 0xFF00) | data;
281 vram_addr = tram_addr;
282 address_latch = 0;
283 }
284 break;
285 case 0x0007: // PPU Data
286 // LAUGHTALE_ENGINR_LOG_INFO("ppu addr: " << datatoHexString(vram_addr.reg, 4))
